194inline size_t dotenv::find_var_start(const std::string& str, size_t pos, std::string& start_tag)
195{
196 size_t p1 = str.find('$',pos);
197 size_t p2 = str.find("${",pos);
198 size_t pos_var = (std::min)(p1,p2);
199 if(pos_var != std::string::npos) start_tag = (pos_var == p2)? "${":"$";
200 return pos_var;
201}

213inline size_t dotenv::find_var_end(const std::string& str, size_t pos, const std::string& start_tag)
214{
215 char end_tag = (start_tag == "${")? '}':' ';
216 size_t pos_end = str.find(end_tag,pos);
217 // special case when $VARIABLE is at end of str with no trailing whitespace
218 if(pos_end == std::string::npos && end_tag==' ') pos_end = str.length();
219 return pos_end;
220}

253inline std::pair<std::string, bool> dotenv::resolve_vars(size_t iline, const std::string& str)
254{
255 std::string resolved;
256
257 size_t pos = 0;
258 size_t pre_pos = pos;
259 size_t nvar = 0;
260
261 bool finished=false;
262 while(!finished)
263 {
264 // look for start of variable expression after pos
265 std::string start_tag;
266 pos = find_var_start(str,pos,start_tag);
267 if(pos != std::string::npos)
268 {
269 // a variable definition detected
270 nvar++;
271
272 // keep start of variable expression
273 size_t pos_start = pos;
274
275 size_t lstart = start_tag.length(); // length of start tag
276 size_t lend = (lstart>1)? 1 : 0; // length of end tag
277
278 // add substring since last variable
279 resolved += str.substr(pre_pos,pos-pre_pos);
280
281 // look for end of variable expression
282 pos = find_var_end(str,pos,start_tag);
283 if(pos != std::string::npos)
284 {
285 // variable name with decoration
286 std::string var = str.substr(pos_start,pos-pos_start+1);
287
288 // variable name without decoration
289 std::string env_var = var.substr(lstart,var.length()-lstart-lend);
290
291 // remove possible whitespace at the end
292 rtrim(env_var);
293
294 // evaluate environment variable
295 if(const char* env_str = std::getenv(env_var.c_str()))
296 {
297 resolved += env_str;
298 nvar--; // decrement to indicate variable resolved
299 }
300 else
301 {
302 // could not resolve the variable, so don't decrement
303 std::cout << "dotenv: Variable " << var << " is not defined on line " << iline << std::endl;
304 }
305
306 // skip end tag
307 pre_pos = pos+lend;
308 }
309 }
310 else {
311 // no more variables
312 finished = true;
313 }
314 }
315
316 // add possible trailing non-whitespace after last variable
317 if(pre_pos < str.length())
318 {
319 resolved += str.substr(pre_pos);
320 }
321
322 // nvar must be 0, or else we have an error
323 return std::make_pair(resolved,(nvar==0));
324}

326inline void dotenv::do_init(int flags, const char* filename)
327{
328 std::ifstream file;
329 std::string line;
330
331 file.open(filename);
332
333 if (file)
334 {
335 unsigned int i = 1;
336
337 while (getline(file, line))
338 {
339 const auto len = line.length();
340 if (len == 0 || line[0] == '#') {
341 continue;
342 }
343
344 const auto pos = line.find("=");
345
346 if (pos == std::string::npos) {
347 std::cout << "dotenv: Ignoring ill-formed assignment on line "
348 << i << ": '" << line << "'" << std::endl;
349 } else {
350 auto name = trim_copy(line.substr(0, pos));
351 auto line_stripped = strip_quotes(trim_copy(line.substr(pos + 1)));
352
353 // resolve any contained variable expressions in 'line_stripped'
354 auto p = resolve_vars(i,line_stripped);
355 bool ok = p.second;
356 if(!ok) {
357 std::cout << "dotenv: Ignoring ill-formed assignment on line "
358 << i << ": '" << line << "'" << std::endl;
359 }
360 else {
361
362 // variable resolved ok, set as environment variable
363 const auto& val = p.first;
364 setenv(name.c_str(), val.c_str(), ~flags & dotenv::Preserve);
365 }
366 }
367 ++i;
368 }
369 }
370}

372inline std::string dotenv::strip_quotes(const std::string& str)
373{
374 const std::size_t len = str.length();
375
376 if (len < 2)
377 return str;
378
379 const char first = str[0];
380 const char last = str[len - 1];
381
382 if (first == last && ('"' == first || '\'' == first))
383 return str.substr(1, len - 2);
384
385 return str;
386}
